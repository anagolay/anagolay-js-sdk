  ## Install
  
  ```sh
  # install latest version
  yarn add @sensio/op-sn-multihash

  # or specific version
  yarn add @sensio/op-sn-multihash@0.3.0
  ```
  
  ## Usage
  
  ```ts
  import snMultihash from '@sensio/op-sn-multihash'


  const data = new U8intArray(7)
  await snMultihash(data)
  ```
