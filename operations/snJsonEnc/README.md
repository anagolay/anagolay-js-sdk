  ## Install
  
  ```sh
  # install latest version
  yarn add @sensio/op-sn-json-enc

  # or specific version
  yarn add @sensio/op-sn-json-enc@0.3.0
  ```
  
  ## Usage
  
  ```ts
  import snJsonEnc from '@sensio/op-sn-json-enc'


  const data = new U8intArray(7)
  await snJsonEnc(data)
  ```
